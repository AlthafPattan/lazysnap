Usage Examples

Three lines to install. One function call to use.

Vanilla JS / @lazysnap/core 
import { observe, configure } from "@lazysnap/core";

// Set global defaults once
configure({ rootMargin: "0px 0px 300px 0px" });

const img = document.querySelector("img");

const cleanup = observe(img, {
  src: "/photos/hero.jpg",
  placeholder: "/photos/hero-lqip.jpg",
  srcset: "hero-480.jpg 480w, hero-1200.jpg 1200w",
  onLoad: () => console.log("loaded!"),
});

// On cleanup / route change:
cleanup();
React / @lazysnap/react 
import { LazyImage } from "@lazysnap/react";

function ProductCard() {
  return (
    <LazyImage
      src="/images/product.jpg"
      placeholder="/images/product-lqip.jpg"
      srcset="product-480.jpg 480w, product-1200.jpg 1200w"
      sizes="(max-width: 600px) 100vw, 400px"
      alt="Product"
      loadedClassName="is-loaded"
      width={800}
      height={600}
    />
  );
}
Angular / @lazysnap/angular 
// component.ts
import { LazysnapDirective } from "@lazysnap/angular";

@Component({
  standalone: true,
  imports: [LazysnapDirective],
  template: `
    <img
      lazysnap
      [lazysnapSrc]="imageUrl"
      [lazysnapPlaceholder]="lqipUrl"
      [lazysnapSrcset]="srcset"
      (lazysnapLoaded)="onLoaded($event)"
      alt="Product image"
    />
  `
})
export class ProductCardComponent {}
React hook / useLazyImage 
import { useLazyImage } from "@lazysnap/react";

function HeroImage() {
  const { ref, isLoaded, state } = useLazyImage({
    src: "/hero.jpg",
    placeholder: "/hero-lqip.jpg",
    transitionDuration: 600,
    onVisible: () =>
      analytics.track("hero_visible"),
  });

  return (
    <img
      ref={ref}
      alt="Hero"
      className={isLoaded ? "loaded" : ""}
    />
  );
}
